112 __attribute__ ((__always_inline__))
113 void wdt_interrupt_enable (const uint8_t value)
114 {
115  if (_SFR_IO_REG_P (_WD_CONTROL_REG))
116  {
117  __asm__ __volatile__ (
118  "in __tmp_reg__,__SREG__" "\n\t"
119  "cli" "\n\t"
120  "wdr" "\n\t"
121  "out %0, %1" "\n\t"
122  "out __SREG__,__tmp_reg__" "\n\t"
123  "out %0, %2" "\n\t"
124  : /* no outputs */
125  : "I" (_SFR_IO_ADDR(_WD_CONTROL_REG)),
126  "r" ((uint8_t)(_BV(_WD_CHANGE_BIT) | _BV(WDE))),
127  "r" ((uint8_t) ((value & 0x08 ? _WD_PS3_MASK : 0x00) |
128  _BV(WDIF) | _BV(WDIE) | (value & 0x07)) )
129  : "r0"
130  );
131  }
132  else
133  {
134  __asm__ __volatile__ (
135  "in __tmp_reg__,__SREG__" "\n\t"
136  "cli" "\n\t"
137  "wdr" "\n\t"
138  "sts %0, %1" "\n\t"
139  "out __SREG__,__tmp_reg__" "\n\t"
140  "sts %0, %2" "\n\t"
141  : /* no outputs */
142  : "n" (_SFR_MEM_ADDR(_WD_CONTROL_REG)),
143  "r" ((uint8_t)(_BV(_WD_CHANGE_BIT) | _BV(WDE))),
144  "r" ((uint8_t) ((value & 0x08 ? _WD_PS3_MASK : 0x00) |
145  _BV(WDIF) | _BV(WDIE) | (value & 0x07)) )
146  : "r0"
